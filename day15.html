from collections import deque
class Solution:
    def findBottomLeftValue(self, root):
        if root is None:
            return 0
        
        # dequestack, FIFO
        queue = deque()
        queue.append(root)
        result = 0
        while queue:
            size = len(queue)
            for i in range(size):
                node = queue.popleft()
                if i == 0:
                    result = node.val
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        return result
